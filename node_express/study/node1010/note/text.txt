import org.springframework.stereotype.Component;

@Component
public class TimeLoopTask {

    @Scheduled(fixedRate = 5000) // 매 5초마다 실행
    public void executeTask() {
        System.out.println("Task is executing every 5 seconds");
    }
}

//작업중인 파일목록 보여줌
const fs = require("fs");

// 파일내용 보여줌 readFile
//utf8 < 인코딩 방식
fs.readFile("text.txt","utf8",(err,data)=>{
    if(err){
        console.log(err);
        return;
    }
    //console.log(data.toString());
    console.log(data);
});

//파일목록 보여줌 readdir
const fs = require("fs");
fs.readdir("./",(err,files)=>{
    if(err){
        console.log("오류!",err);
        return;
    }
    console.log(files);
});
